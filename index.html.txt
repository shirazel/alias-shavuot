    <script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    
    <!-- Babel for JSX -->
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <style>
        @import url('https://fonts.googleapis.com/css2?family=Assistant:wght@300;400;500;600;700&display=swap');
        
        body {
            font-family: 'Assistant', sans-serif;
            background-color: #fafaf9;
            background-image: radial-gradient(#d1fae5 2px, transparent 2px);
            background-size: 30px 30px;

        .board-grid {
            display: grid;
            grid-template-columns: repeat(10, 1fr);
            grid-template-rows: repeat(10, 1fr);
        }
        
        .thief-space {
            background-color: #ecfdf5;
            border: 1px solid #10b981;
        }

        .modal-overlay {
            background-color: rgba(0, 0, 0, 0.4);
            backdrop-filter: blur(4px);
            z-index: 1000;
        }

        /* עיצוב קישור הוראות עדין וקטן כפי שביקשת */
        .instruction-link {
            color: #10b981;
            font-size: 13px;
            font-weight: 400;
            text-decoration: underline;
            background: none;
            border: none;
            cursor: pointer;
            transition: color 0.2s;
        }
        .instruction-link:hover {
            color: #064e3b;

<body>
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useCallback, useRef, useMemo } = React;

        const ICONS = ['🧀', '🌾', '🍇', '🍷', '🍯', '🕊️', '🚜', '🧺', '🐮', '🥞'];

        const WORDS_POOL = [
            "חג שבועות", "ביכורים", "טנא", "הר סיני", "מתן תורה", "עשרת הדיברות", "לוחות הברית", "משה רבנו",
            "עוגת גבינה", "חלב", "פרה", "חקלאות", "מים", "זר פרחים", "לבן", "קציר",
            "חיטה", "שעורה", "גפן", "תאנה", "רימון", "זית", "דבש", "שבעת המינים",
            "מגילת רות", "נעמי", "בועז", "אדמה", "שדה", "טרקטור", "מגל", "פירות",
            "תנובה", "כבשה", "יוגורט", "חמאה", "מעדן", "גלידה", "חג הקציר", "חג הביכורים",
            "עצרת", "תיקון ליל שבועות", "בלינצ'ס", "פשטידה", "כרם", "פרדס", "מעיין", "שמש",
            "תבואה", "אלומות", "אוהל", "מדבר", "לחם", "בגדים לבנים", "תפילה", "ארץ זבת חלב",
            "ירושלים", "בית המקדש", "ענבים", "שמן זית", "דבורה", "כוורת", "כד חלב", "ספירת העומר",
            "פיצה", "טוסט", "רובה מים", "השפרצה", "בריכה", "חקלאי", "קצפת", "שבועה",
            "גבינה צהובה", "גבינה לבנה", "רפת", "עטינים", "חציר", "תורה", "אותיות", "קריאה",
            "מגילה", "קציר חיטים", "מחרשה", "אפייה", "קמח", "בצק", "שמרים", "פחמימות"
        ];

        const SYNC_KEY = 'alias_shavuot_sync_vFinal';
        const channel = new BroadcastChannel('shavuot_channel_final');

        const App = () => {
            const [isProjector] = useState(() => window.location.search.includes('mode=projector'));
            const [showInstructions, setShowInstructions] = useState(false);
            const [gameState, setGameState] = useState('SETUP');
            const [teams, setTeams] = useState([
                { id: 1, name: 'קבוצה 1', icon: ICONS[0], score: 1 }, 
                { id: 2, name: 'קבוצה 2', icon: ICONS[1], score: 1 }
            ]);
            const [activeTeamIdx, setActiveTeamIdx] = useState(0);
            const [timeLeft, setTimeLeft] = useState(60);
            const [currentWords, setCurrentWords] = useState([]);
            const [points, setPoints] = useState({});
            const [penalties, setPenalties] = useState(0);
            const [cardsCount, setCardsCount] = useState(0);

            const activeTeam = teams[activeTeamIdx] || teams[0];
            const targetIdx = (activeTeam.score - 1) % 8;
            const isRobbery = activeTeam.score % 8 === 0;

            // שידור מצב לכל החלונות
            const broadcast = useCallback((data) => {
                if (isProjector) return;
                const state = {
                    gameState, teams, activeTeamIdx, timeLeft, currentWords, 
                    points, penalties, cardsCount, ...data
                };
                channel.postMessage(state);
                localStorage.setItem(SYNC_KEY, JSON.stringify(state));
            }, [isProjector, gameState, teams, activeTeamIdx, timeLeft, currentWords, points, penalties, cardsCount]);

            // קבלת עדכונים
            useEffect(() => {
                const handleSync = (s) => {
                    if (!s) return;
                    setGameState(s.gameState); setTeams(s.teams); setActiveTeamIdx(s.activeTeamIdx);
                    setTimeLeft(s.timeLeft); setCurrentWords(s.currentWords);
                    setPoints(s.points); setPenalties(s.penalties); setCardsCount(s.cardsCount);
                };
                channel.onmessage = (e) => handleSync(e.data);
                window.onstorage = (e) => { if (e.key === SYNC_KEY) handleSync(JSON.parse(e.newValue)); };
                if (isProjector) {
                    const saved = localStorage.getItem(SYNC_KEY);
                    if (saved) handleSync(JSON.parse(saved));
                }
            }, [isProjector]);

            // טיימר
            useEffect(() => {
                let timer;
                if (!isProjector && gameState === 'PLAYING' && !isRobbery && timeLeft > 0) {
                    timer = setInterval(() => {
                        setTimeLeft(prev => {
                            const nt = prev - 1;
                            broadcast({ timeLeft: nt });
                            return nt;
                        });
                    }, 1000);
                } else if (!isProjector && gameState === 'PLAYING' && !isRobbery && timeLeft === 0) {
                    setGameState('SUMMARY'); broadcast({ gameState: 'SUMMARY' });
                }
                return () => clearInterval(timer);
            }, [gameState, timeLeft, isProjector, isRobbery, broadcast]);

            const startTurn = () => {
                const initialPoints = {};
                teams.forEach(t => initialPoints[t.id] = 0);
                const words = [...WORDS_POOL].sort(() => 0.5 - Math.random()).slice(0, 8);
                setGameState('PLAYING'); setTimeLeft(60); setPoints(initialPoints); setPenalties(0); setCardsCount(0); setCurrentWords(words);
                broadcast({ gameState: 'PLAYING', timeLeft: 60, points: initialPoints, penalties: 0, cardsCount: 0, currentWords: words });
            };

            const handleAnswer = (id) => {
                const nextPoints = { ...points };
                let nextPenalties = penalties;
                const nextCount = cardsCount + 1;
                if (id === 'SKIP') nextPenalties++;
                else nextPoints[id] = (nextPoints[id] || 0) + 1;

                if (isRobbery && nextCount >= 5) {
                    setGameState('SUMMARY');
                    broadcast({ gameState: 'SUMMARY', points: nextPoints, penalties: nextPenalties, cardsCount: nextCount });
                } else {
                    const words = [...WORDS_POOL].sort(() => 0.5 - Math.random()).slice(0, 8);
                    setPoints(nextPoints); setPenalties(nextPenalties); setCardsCount(nextCount); setCurrentWords(words);
                    broadcast({ points: nextPoints, penalties: nextPenalties, cardsCount: nextCount, currentWords: words });
                }
            };

            const finalizeTurn = () => {
                const nextTeams = teams.map(t => {
                    const p = points[t.id] || 0;
                    const pen = t.id === activeTeam.id ? penalties : 0;
                    return { ...t, score: Math.max(1, t.score + p - pen) };
                });
                const nextIdx = (activeTeamIdx + 1) % nextTeams.length;
                setTeams(nextTeams);
                if (nextTeams.some(t => t.score >= 100)) {
                    setGameState('WINNER'); broadcast({ teams: nextTeams, gameState: 'WINNER' });
                } else {
                    setActiveTeamIdx(nextIdx); setGameState('BOARD');
                    broadcast({ teams: nextTeams, gameState: 'BOARD', activeTeamIdx: nextIdx });
                }
            };

            if (isProjector) {
                return (
                    <div className="h-screen overflow-hidden flex flex-col p-4 bg-gray-50 text-center">
                        <div className="flex flex-col h-full gap-4">
                            {gameState === 'PLAYING' && (
                                <div className="mx-auto bg-white border border-emerald-100 rounded-full py-4 px-12 shadow-sm">
                                    {isRobbery ? <div className="text-3xl font-medium text-emerald-600 animate-pulse">🚨 סיבוב גנב! כולם מנחשים 🚨</div> : <div className="text-6xl font-bold text-emerald-800">{timeLeft}</div>}
                                </div>
                            )}
                            <div className="flex-grow bg-white rounded-[2rem] p-6 shadow-sm border border-emerald-50 flex flex-col">
                                <h2 className="text-3xl font-bold text-emerald-800 mb-4">
                                    {gameState === 'SETUP' ? 'ברוכים הבאים! 🌾' : (gameState === 'WINNER' ? 'יש מנצחים! 🏆' : `תור: ${activeTeam.icon} ${activeTeam.name}`)}
                                </h2>
                                <div className="board-grid bg-gray-50 p-2 rounded-2xl flex-grow">
                                    {Array.from({ length: 100 }).map((_, i) => {
                                        const s = i + 1;
                                        const here = teams.filter(t => (gameState === 'SUMMARY' ? Math.max(1, t.score + (points[t.id] || 0) - (t.id === activeTeam.id ? penalties : 0)) : t.score) === s);
                                        return (
                                            <div key={s} className={`border rounded flex items-center justify-center relative ${s % 8 === 0 ? 'thief-space' : 'bg-white'}`}>
                                                <span className="absolute top-0.5 right-1 text-[1vmin] text-gray-300">{(s-1)%8+1}</span>
                                                <div className="flex flex-wrap justify-center">{here.map(t => <span key={t.id} className="text-[4vmin]">{t.icon}</span>)}</div>
                                            </div>
                                        );
                                    })}
                                </div>
                                <div className="mt-4 opacity-50 text-[12px] font-medium text-emerald-900">חג שבועות שמח - באהבה פסג"ה חדרה 🌾</div>
                            </div>
                        </div>
                    </div>
                );
            }

            return (
                <div className="min-h-screen flex flex-col items-center justify-center p-4 bg-gray-50 relative">
                    {showInstructions && (
                        <div className="fixed inset-0 modal-overlay flex items-center justify-center p-4">
                            <div className="bg-white rounded-3xl p-8 max-w-md w-full shadow-2xl text-right">
                                <h2 className="text-2xl font-bold text-emerald-800 mb-4 border-b pb-2">איך משחקים?</h2>
                                <div className="space-y-3 text-gray-600 text-sm">
                                    <p>• מסבירים את המילה בלי להגיד אותה או את שורשה.</p>
                                    <p>• משבצת ירוקה כהה = "גנב". כולם מנחשים ואין טיימר (עד 5 מילים).</p>
                                    <p>• לחצו על כפתור המקרן בפינה וגררו את החלון החדש למקרן הכיתה.</p>
                                </div>
                                <button onClick={() => setShowInstructions(false)} className="w-full mt-6 bg-emerald-500 text-white font-bold py-3 rounded-xl shadow-lg">הבנתי!</button>
                            </div>
                        </div>
                    )}

                    <button onClick={() => window.open(window.location.href.split('?')[0] + "?mode=projector", '_blank')} className="absolute top-4 left-4 bg-white text-emerald-600 border border-emerald-100 px-4 py-1.5 rounded-full font-medium text-[11px] shadow-sm hover:bg-emerald-50 active:scale-95 transition-all">📽️ פתח חלון למקרן</button>
                    
                    {gameState === 'SETUP' && (
                        <div className="bg-white p-10 rounded-3xl card-shadow max-w-sm w-full text-center border border-emerald-50">
                            <h1 className="text-4xl font-bold text-emerald-800 mb-2">אליאס שבועות</h1>
                            <button onClick={() => setShowInstructions(true)} className="instruction-link mb-8 block mx-auto">הוראות משחק</button>
                            <div className="space-y-3">
                                {teams.map((t, i) => (
                                    <div key={t.id} className="flex items-center gap-2">
                                        <span className="text-xl">{t.icon}</span>
                                        <input type="text" value={t.name} onChange={(e) => { const nt = [...teams]; nt[i].name = e.target.value; setTeams(nt); broadcast({ teams: nt }); }} className="flex-grow border rounded-lg p-2 font-bold focus:outline-none" />
                                        {teams.length > 2 && i === teams.length - 1 && <button onClick={() => setTeams(teams.filter(tm => tm.id !== t.id))} className="text-red-300 px-2">🗑️</button>}
                                    </div>
                                ))}
                                <button onClick={() => setTeams([...teams, { id: Date.now(), name: `קבוצה ${teams.length + 1}`, icon: ICONS[teams.length % 10], score: 1 }])} className="w-full border border-dashed border-emerald-200 text-emerald-600 py-2 rounded-xl text-[11px] font-bold">+ הוסף קבוצה</button>
                                <button onClick={() => { setGameState('BOARD'); broadcast({ gameState: 'BOARD' }); }} className="w-full mt-6 bg-emerald-500 text-white font-bold text-lg py-4 rounded-2xl shadow-lg">בואו נשחק!</button>
                            </div>
                        </div>
                    )}

                    {gameState === 'BOARD' && (
                        <div className="max-w-5xl w-full flex flex-col xl:flex-row gap-6">
                            <div className="xl:flex-[2.5] bg-white rounded-3xl p-6 shadow-sm border border-emerald-50">
                                <div className="flex justify-between items-center mb-6">
                                    <button onClick={() => setGameState('SETUP')} className="text-gray-400 text-[11px] hover:text-emerald-500">⚙️ הגדרות</button>
                                    <h2 className="text-xl font-bold text-emerald-800">מפת המשחק</h2>
                                    <button onClick={() => setShowInstructions(true)} className="instruction-link">הוראות</button>
                                </div>
                                <div className="board-grid bg-gray-50 p-2 rounded-2xl h-[450px]">
                                    {Array.from({ length: 100 }).map((_, i) => (
                                        <div key={i+1} className={`border rounded flex items-center justify-center relative ${(i+1)%8===0 ? 'thief-space' : 'bg-white'}`}>
                                            {teams.filter(t => t.score === (i+1)).map(t => <span key={t.id} className={`text-xl ${t.id === activeTeam.id ? 'animate-bounce' : ''}`}>{t.icon}</span>)}
                                        </div>
                                    ))}
                                </div>
                            </div>
                            <div className="xl:flex-1 bg-white p-8 rounded-3xl shadow-sm text-center flex flex-col justify-center border border-emerald-50">
                                <h3 className="text-3xl font-bold mb-4">{activeTeam.icon} {activeTeam.name}</h3>
                                {isRobbery ? <div className="bg-emerald-600 text-white p-4 rounded-xl mb-6 font-bold animate-pulse">🚨 שוד! 5 קלפים לכולם</div> : <div className="bg-emerald-50 p-6 rounded-2xl mb-6 border border-emerald-100 shadow-inner"><p className="text-xs text-gray-500 uppercase">המשימה שלכם:</p><span className="text-6xl font-bold text-emerald-600">{targetIdx + 1}</span></div>}
                                <button onClick={startTurn} className="w-full bg-emerald-500 text-white font-bold text-xl py-6 rounded-2xl shadow-md active:scale-95">התחל תור!</button>
                            </div>
                        </div>
                    )}

                    {gameState === 'PLAYING' && (
                        <div className="max-w-sm w-full">
                            <div className={`flex justify-between items-center p-4 rounded-2xl bg-white shadow-sm border-b-4 mb-4 ${isRobbery ? 'border-emerald-600' : 'border-emerald-400'}`}>
                                <span className="font-bold text-lg">{activeTeam.icon} {activeTeam.name}</span>
                                <span className="text-4xl font-bold text-emerald-800">{isRobbery ? '🚨' : timeLeft}</span>
                            </div>
                            <div className="bg-white rounded-[2rem] p-6 shadow-xl flex flex-col gap-3">
                                {currentWords.map((word, i) => (
                                    <div key={i} className={`flex items-center p-3 rounded-xl transition-all ${i === targetIdx ? 'bg-emerald-500 text-white scale-105 shadow-lg font-bold' : 'opacity-20 blur-[0.5px]'}`}>
                                        <div className={`w-8 h-8 rounded-full flex items-center justify-center font-bold ml-4 ${i === targetIdx ? 'bg-white text-emerald-600' : 'bg-gray-100'}`}>{i + 1}</div>
                                        <div className="text-2xl font-semibold tracking-tight">{word}</div>
                                    </div>
                                ))}
                            </div>
                            <div className="grid grid-cols-2 gap-3 mt-6">
                                <button onClick={() => handleAnswer('SKIP')} className="bg-white text-rose-400 py-4 rounded-xl font-bold border border-rose-100 text-xs">❌ דילוג (-1)</button>
                                {!isRobbery ? (
                                    <button onClick={() => handleAnswer(activeTeam.id)} className="bg-emerald-500 text-white py-4 rounded-xl font-bold text-xl shadow-md">✅ נכון!</button>
                                ) : (
                                    <div className="col-span-2 grid grid-cols-2 gap-2 mt-2">
                                        {teams.map(t => <button key={t.id} onClick={() => handleAnswer(t.id)} className="bg-emerald-50 border border-emerald-100 text-emerald-700 py-3 rounded-lg text-xs font-bold shadow-sm">{t.icon} {t.name}</button>)}
                                    </div>
                                )}
                            </div>
                        </div>
                    )}

                    {gameState === 'SUMMARY' && (
                        <div className="bg-white p-12 rounded-[2.5rem] card-shadow max-w-sm w-full text-center border border-emerald-50">
                            <h2 className="text-3xl font-bold text-emerald-800 mb-6 tracking-tighter">סיכום התור ⏰</h2>
                            <div className="bg-gray-50 p-6 rounded-2xl mb-8 space-y-3 text-right font-bold shadow-inner">
                                {isRobbery ? teams.map(t => (points[t.id] || 0) > 0 && <div key={t.id} className="flex justify-between"><span>{t.icon} {t.name}</span><span className="text-emerald-600">+{points[t.id]}</span></div>) : <div className="flex justify-between text-xl"><span>נכונות:</span><span className="text-emerald-600">{points[activeTeam.id] || 0}</span></div>}
                                <div className="flex justify-between text-rose-500"><span>דילוגים:</span><span>{penalties}</span></div>
                            </div>
                            <button onClick={finalizeTurn} className="w-full bg-emerald-500 text-white font-bold text-lg py-5 rounded-xl shadow-lg active:scale-95 shadow-emerald-100">חזרה למפה 🗺️</button>
                        </div>
                    )}

                    {gameState === 'WINNER' && (
                        <div className="text-center bg-white p-12 rounded-[3rem] card-shadow border border-emerald-50">
                            <h1 className="text-5xl font-bold text-emerald-800 mb-8">יש מנצחים! 🏆</h1>
                            <button onClick={() => { setTeams(teams.map(t=>({...t,score:1}))); setGameState('SETUP'); broadcast({ teams: teams.map(t=>({...t,score:1})), gameState: 'SETUP' }); }} className="bg-emerald-600 text-white font-bold py-6 px-16 rounded-full text-2xl shadow-xl active:scale-95">משחק חדש 🔄</button>
                        </div>
                    )}
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
